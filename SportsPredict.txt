        function refreshSBOStream() {
            const iframe = document.getElementById('sboLiveStream');
            iframe.src = iframe.src;
            
            // Simulasi update jadwal pertandingan
            const matchList = document.getElementById('sboMatchList');
            const matches = matchList.querySelectorAll('.sbo-match-item');
            
            matches.forEach(match => {
                const scoreElement = match.querySelector('.sbo-score');
                const timeElement = match.querySelector('.sbo-time');
                
                if (scoreElement.textContent === '-') {
                    // Jika pertandingan belum dimulai, tidak ada perubahan
                    return;
                }
                
                // Update skor secara acak
                const currentScore = scoreElement.textContent.split(' - ');
                let homeScore = parseInt(currentScore[0]);
                let awayScore = parseInt(currentScore[1]);
                
                // 20% kemungkinan terjadi gol
                if (Math.random() < 0.2) {
                    if (Math.random() < 0.5) {
                        homeScore += 1;
                    } else {
                        awayScore += 1;
                    }
                    scoreElement.textContent = `${homeScore} - ${awayScore}`;
                }
                
                // Update waktu pertandingan
                if (timeElement.textContent.includes("'")) {
                    let minute = parseInt(timeElement.textContent);
                    if (minute < 90) {
                        minute += 5;
                        timeElement.textContent = `${minute}'`;
                    } else {
                        timeElement.textContent = 'FT';
                    }
                }
            });
            
            // Tampilkan notifikasi
            showNotification('SBO Virtual Live Stream telah diperbarui!', 'info');
        }

        function generateTable() {
            let input = document.getElementById("inputData").value.trim();
            if (!input) {
                showNotification("Masukkan data prediksi terlebih dahulu!", "error");
                return;
            }
            
            let lines = input.split("\n").filter(line => line.trim() !== "");
            let tableHtml = "";
            let currentLeague = "";

            for (let i = 0; i < lines.length; i++) {
                let line = lines[i].trim();
                
                // Deteksi apakah baris ini adalah nama liga
                if (!line.includes("vs") && !line.includes("VS") && line.length > 0 && !line.match(/^\d{2}\/\d{2} \d{2}:\d{2} WIB/)) {
                    if (currentLeague) {
                        tableHtml += "</tbody></table>";
                    }
                    currentLeague = line;
                    tableHtml += `
                    <div class="table-container">
                        <table class="league-table">
                            <tr><th colspan="3" class="league-name">${currentLeague}</th></tr>
                            <tr>
                                <th>Tanggal & Waktu</th>
                                <th>Pertandingan</th>
                                <th>Skor</th>
                            </tr>
                            <tbody>
                    `;
                } else {
                    // Deteksi apakah baris ini adalah pertandingan
                    let matchPattern = /^(\d{2}\/\d{2} \d{2}:\d{2} WIB)\s+(.+?)\s+[vV][sS]\s+(.+?)\s+(\d+-\d+)$/;
                    let match = line.match(matchPattern);
                    if (match) {
                        let date = match[1];
                        let team1 = match[2].trim();
                        let team2 = match[3].trim();
                        let score = match[4].replace("-", " : ");
                        tableHtml += `
                            <tr>
                                <td>${date}</td>
                                <td>${team1} VS ${team2}</td>
                                <td>${score}</td>
                            </tr>
                        `;
                    }
                }
            }

            if (currentLeague) {
                tableHtml += "</tbody></table></div>";
            }
            
            document.getElementById("outputTable").innerHTML = tableHtml;
            document.getElementById("htmlCode").style.display = "none";
            showNotification("Tabel prediksi berhasil digenerate!", "success");
        }

        // Fungsi untuk menyalin hasil jadwal
        function copyToClipboard() {
            const outputElement = document.getElementById('output');
            if (!outputElement || !outputElement.textContent) {
                showNotification("Tidak ada hasil untuk disalin. Proses jadwal terlebih dahulu!", "error");
                return;
            }
            
            const result = outputElement.textContent;
            navigator.clipboard.writeText(result)
                .then(() => {
                    showNotification("Hasil jadwal berhasil disalin ke clipboard!", "success");
                })
                .catch(err => {
                    console.error('Gagal menyalin teks: ', err);
                    // Fallback untuk browser yang tidak mendukung clipboard API
                    const textArea = document.createElement('textarea');
                    textArea.value = result;
                    document.body.appendChild(textArea);
                    textArea.select();
                    document.execCommand('copy');
                    document.body.removeChild(textArea);
                    showNotification("Hasil jadwal berhasil disalin ke clipboard!", "success");
                });
        }
        
         // Fungsi untuk konversi skor
        function konversiSkor() {
            let input = document.getElementById("inputSkor").value.trim();
            if (!input) {
                showNotification("Masukkan data pertandingan terlebih dahulu!", "error");
                return;
            }
            
            let lines = input.split("\n").filter(line => line.trim() !== "");
            let output = "";
            let currentLeague = "";

            for (let i = 0; i < lines.length; i++) {
                let line = lines[i].trim();
                
                // Deteksi apakah baris ini adalah nama liga
                if (!line.includes("vs") && !line.includes("VS") && line.length > 0 && !line.match(/^\d{2}\/\d{2} \d{2}:\d{2} WIB/)) {
                    if (currentLeague) {
                        output += "\n";
                    }
                    currentLeague = line;
                    output += currentLeague + "\n\n";
                } else {
                    // Deteksi apakah baris ini adalah pertandingan
                    let matchPattern = /^(\d{2}\/\d{2} \d{2}:\d{2} WIB)\s+(.+?)\s+[vV][sS]\s+(.+?)\s+(\d+)\s*[:|-]\s*(\d+)$/;
                    let match = line.match(matchPattern);
                    if (match) {
                        let date = match[1];
                        let team1 = match[2].trim();
                        let team2 = match[3].trim();
                        let score1 = match[4];
                        let score2 = match[5];
                        
                        // Konversi skor ke format "1-0"
                        let convertedScore = `${score1}-${score2}`;
                        
                        output += `${date} ${team1} vs ${team2} ${convertedScore}\n`;
                    } else {
                        // Jika tidak sesuai pola, tampilkan baris asli
                        output += line + "\n";
                    }
                }
            }

            document.getElementById("outputKonversi").innerHTML = `<pre style="color: #FFD700; font-family: 'Courier New', monospace; white-space: pre-wrap;">${output}</pre>`;
            showNotification("Konversi skor berhasil!", "success");
        }
